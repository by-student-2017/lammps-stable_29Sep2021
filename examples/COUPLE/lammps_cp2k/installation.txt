## Note
D drive = "/mnt/d". Usually "/mnt/d" => "$HOME" or "~"

## Installation (NWCHEM)
1. sudo apt update
2. sudo apt -y install cp2k
3. which cp2k
(/usr/bin/cp2k)

## Use file (No ZMQ)
#------------------------------------------
## Build LAMMPS with its MESSAGE package installed (No ZMQ)
0. sudo apt update
  sudo apt -y install libzmq3-dev
1. cd /mnt/d
2. git clone https://github.com/by-student-2017/lammps-stable_29Sep2021.git
3. cd /mnt/d/lammps-stable_29Sep2021/lib/message
4. python2 Install.py -m
5. cd /mnt/d/lammps-stable_29Sep2021/src
6. make yes-message
7. make mpi

## Build the CSlib in a form usable by the vasp_wrapper.py script (No ZMQ)
1. cd /mnt/d/lammps-stable_29Sep2021/lib/message/cslib/src
2. make shlib zmq=no

## Environment settings
1. echo '# lammps-couple environment settings' >> ~/.bashrc
2. echo 'export PYTHONPATH=${PYTHONPATH}:/mnt/d/lammps-stable_29Sep2021/lib/message/cslib/src' >> ~/.bashrc
3. echo 'export LD_LIBRARY_PATH=${LD_LIBRARY_PATH}:/mnt/d/lammps-stable_29Sep2021/lib/message/cslib/src' >> ~/.bashrc
4. bash

## Usage (No ZMQ) (Plane wave)
1. cd /mnt/d/lammps-stable_29Sep2021/examples/COUPLE/lammps_nwchem/planewave
2. mkdir cfg
3. export OMP_NUM_THREADS=1
4. mpirun -np 1 /mnt/d/lammps-stable_29Sep2021/src/lmp_mpi -v mode file -in in.client.lmp & python2 ./../cp2k_wrap.py file pw cp2k_tmp.inp &
5. (use kill command)

## Usage (No ZMQ) (Atomic orbital basis)
1. cd /mnt/d/lammps-stable_29Sep2021/examples/COUPLE/lammps_nwchem/ao_basis
2. mkdir cfg
3. export OMP_NUM_THREADS=1
4. mpirun -np 1 /mnt/d/lammps-stable_29Sep2021/src/lmp_mpi -v mode file -in in.client.lmp & python2 ./../cp2k_wrap.py file ao cp2k_tmp.xyz &
5. (use kill command)
